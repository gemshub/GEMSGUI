Instruction:

Add new wizards for Remake/Create module procedure:
(see ComposWizard as example)

1. Create ModuleWizard.ui file using QtDisigner,
   and put it to gems2\vizor\dlg catalog

2. Execute commands in gems2\vizor\dlg catalog:
   uic ModuleWizard.ui -o ModuleWizardData.h
   uic ModuleWizard.ui -impl ModuleWizardData.h -o ModuleWizardData.cpp

5. Added function vfModuleSet to visor_w.cpp and user.h files

#include "dlg/ComposWizard.h"

bool
vfComposSet(QWidget* par, const char * p_key,
            char flgs[6], int size[2], double& r2  )
{
     ComposWizard cdlg( p_key, flgs, size, r2, par );
     if( !cdlg.exec() )
      return false;

    cdlg.getFlags( flgs );
    cdlg.getSizes( size );
    r2 = cdlg.getR2();

    return true;
}

6. Added function MakeQuery() to m_module.cpp/.h files

/* opens window with 'Remake record' parameters
*/
void
TCompos::MakeQuery()
{
//    pImp->MakeQuery();
    const char * p_key;
    char flgs[6];
    int size[2];
    double r2 = bcp->R2;

    p_key  = db->PackKey();
    memcpy( flgs, &bcp->PcIC, 6);
    size[0] = bcp->La;
    size[1] = bcp->Nsd;

    if( !vfComposSet( window(), p_key, flgs, size, r2 ))
        return;   // cancel

    memcpy( &bcp->PcIC, flgs, 6);
    bcp->La = (short)size[0];
    bcp->Nsd = (short)size[1];
    bcp->R2 = r2;

}
